1//!
2//! Xwayland Window Manager module
3//!
4//! Provides an [`X11Wm`] type, which will register itself as a window manager
5//! for a previously spawned Xwayland instances, allowing
6//! backwards-compatibility by seamlessly integrating X11 windows into a wayland
7//! compositor.
8//!
9//! To use this functionality you must first spawn an
10//! [`XWayland`](super::XWayland) instance to attach a [`X11Wm`] to.
11//!
12//! # Associating an [`X11Surface`] with a [`WlSurface`][wayland_server::protocol::wl_surface::WlSurface]
13//!
14//! When an X11 window is created, XWayland will associate it with a
15//! `wl_surface` that it creates via the wayland connection. This happens in two
16//! steps:
17//!
18//!  - Via the [xwayland shell](crate::wayland::xwayland_shell) protocol, a
19//!    serial number is set on the surface, and it is given the
20//!    "xwayland_shell" role.
21//!  - Via the X11 connection, a matching `WL_SURFACE_SERIAL` atom is set on the
22//!    X11 window, which can be queried with
23//!    [`X11Surface::wl_surface_serial()`].
24//!
25//! Note that these two steps can happen in any order.
26//!
27//! # Example
28//!
29//! ```no_run
30//! #  use smithay::wayland::xwayland_shell::{XWaylandShellHandler, XWaylandShellState};
31//! #  use smithay::wayland::selection::SelectionTarget;
32//! #  use smithay::xwayland::{XWayland, XWaylandEvent, X11Wm, X11Surface, XwmHandler, xwm::{XwmId, ResizeEdge, Reorder}};
33//! #  use smithay::utils::{Rectangle, Logical};
34//! #  use std::os::unix::io::OwnedFd;
35//! #  use std::process::Stdio;
36//! #
37//! struct State { /* ... */ }
38//! # impl XWaylandShellHandler for State {
39//! #     fn xwayland_shell_state(&mut self) -> &mut XWaylandShellState {
40//! #         unreachable!()
41//! #     }
42//! # }
43//! impl XwmHandler for State {
44//!     fn xwm_state(&mut self, xwm: XwmId) -> &mut X11Wm {
45//!         // ...
46//! #       unreachable!()
47//!     }
48//!     fn new_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
49//!     fn new_override_redirect_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
50//!     fn map_window_request(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
51//!     fn mapped_override_redirect_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
52//!     fn unmapped_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
53//!     fn destroyed_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
54//!     fn configure_request(&mut self, xwm: XwmId, window: X11Surface, x: Option<i32>, y: Option<i32>, w: Option<u32>, h: Option<u32>, reorder: Option<Reorder>) { /* ... */ }
55//!     fn configure_notify(&mut self, xwm: XwmId, window: X11Surface, geometry: Rectangle<i32, Logical>, above: Option<u32>) { /* ... */ }
56//!     fn resize_request(&mut self, xwm: XwmId, window: X11Surface, button: u32, resize_edge: ResizeEdge) { /* ... */ }
57//!     fn move_request(&mut self, xwm: XwmId, window: X11Surface, button: u32) { /* ... */ }
58//!     fn send_selection(&mut self, xwm: XwmId, selection: SelectionTarget, mime_type: String, fd: OwnedFd) { /* ... */ }
59//! }
60//! #
61//! # let dh = unreachable!();
62//! # let handle: smithay::reexports::calloop::LoopHandle<'static, State> = unreachable!();
63//!
64//! let (xwayland, client) = XWayland::spawn(
65//!     &dh,
66//!     None,
67//!     std::iter::empty::<(String, String)>(),
68//!     true,
69//!     Stdio::null(),
70//!     Stdio::null(),
71//!     |_| (),
72//! )
73//! .expect("failed to start XWayland");
74
75//! let ret = handle.insert_source(xwayland, move |event, _, data| match event {
76//!     XWaylandEvent::Ready {
77//!         x11_socket,
78//!         display_number: _,
79//!     } => {
80//!         let wm = X11Wm::start_wm(
81//!             handle.clone(),
82//!             x11_socket,
83//!             client.clone(),
84//!         )
85//!         .expect("Failed to attach X11 Window Manager");
86//!         
87//!         // store the WM somewhere
88//!     }
89//!     XWaylandEvent::Error => eprintln!("XWayland failed to start!"),
90//! }); if let Err(e) = ret { tracing::error!( "Failed to insert the
91//! XWaylandSource into the event loop: {}", e ); }
92//! ```

250impl StackingDirection {
251    fn pos_comparator(&self, pos: usize, last_pos: usize) -> bool {
252        match self {
253            Self::Downwards => last_pos < pos,
254            Self::Upwards => last_pos > pos,
255        }
256    }

258    fn stack_mode(&self) -> StackMode {
259        match self {
260            Self::Downwards => StackMode::BELOW,
261            Self::Upwards => StackMode::ABOVE,
262        }
263    }

461impl fmt::Debug for IncomingTransfer {

463        f.debug_struct("IncomingTransfer")
464            .field("conn", &"...")
465            .field("token", &self.token)
466            .field("window", &self.window)
467            .field("incr", &self.incr)

474impl IncomingTransfer {
475    fn read_selection_prop(&mut self, reply: GetPropertyReply) {
476        self.source_data.extend(&reply.value)
477    }
478
479    fn write_selection(&mut self, fd: BorrowedFd<'_>) -> std::io::Result<bool> {
480        if self.source_data.is_empty() {
481            return Ok(true);
482        }
483
484        let len = rustix::io::write(fd, &self.source_data)?;
485        self.source_data = self.source_data.split_off(len);
486
487        Ok(self.source_data.is_empty())
488    }
489
490    fn destroy<D>(mut self, handle: &LoopHandle<'_, D>) {
491        if let Some(token) = self.token.take() {
492            handle.remove(token);
493        }
494    }
495}

500        if self.token.is_some() {
501            tracing::warn!(
502                ?self,

530            .field("request", &self.request)

532            .field("flush_property_on_delete", &self.flush_property_on_delete)
533            .finish()
534    }

537impl OutgoingTransfer {
538    fn flush_data(&mut self) -> Result<usize, ReplyOrIdError> {
539        let len = std::cmp::min(self.source_data.len(), INCR_CHUNK_SIZE);
540
541        if len == 0 {
542            // This flush will complete the transfer
543            self.sent_finished = true;
544        }
545
546        let mut data = self.source_data.split_off(len);
547        std::mem::swap(&mut data, &mut self.source_data);
548
549        self.conn.change_property8(
550            PropMode::REPLACE,
551            self.request.requestor,
552            self.request.property,
553            self.request.target,
554            &data,
555        )?;
556        self.conn.flush()?;
557
558        let remaining = self.source_data.len();
559        self.property_set = true;
560        Ok(remaining)
561    }
562
563    fn destroy<D>(mut self, handle: &LoopHandle<'_, D>) {
564        if let Some(token) = self.token.take() {
565            handle.remove(token);
566        }
567    }
568}

573            tracing::warn!(
574                ?self,
575                "OutgoingTransfer freed before being removed from EventLoop"
576            );
577        }
578    }

581impl XWmSelection {
582    fn new(
583        conn: &Arc<RustConnection>,
584        screen: &Screen,
585        atoms: &Atoms,
586        atom: Atom,
587    ) -> Result<Self, ReplyOrIdError> {
588        let window = conn.generate_id()?;
589        conn.create_window(
590            screen.root_depth,
591            window,
592            screen.root,
593            0,
594            0,
595            10,
596            10,
597            0,
598            WindowClass::INPUT_OUTPUT,
599            screen.root_visual,
600            &CreateWindowAux::new().event_mask(EventMask::PROPERTY_CHANGE),
601        )?;
602
603        if atom == atoms.CLIPBOARD {
604            conn.set_selection_owner(window, atoms.CLIPBOARD_MANAGER, x11rb::CURRENT_TIME)?;
605        }
606        conn.xfixes_select_selection_input(
607            window,
608            atom,
609            SelectionEventMask::SET_SELECTION_OWNER
610                | SelectionEventMask::SELECTION_WINDOW_DESTROY
611                | SelectionEventMask::SELECTION_CLIENT_CLOSE,
612        )?;
613        conn.flush()?;
614
615        let selection = match atom {
616            x if x == atoms.CLIPBOARD => SelectionTarget::Clipboard,
617            x if x == atoms.PRIMARY => SelectionTarget::Primary,
618            _ => unreachable!(),
619        };
620
621        debug!(
622            selection_window = ?window,
623            ?selection,
624            ?atom,
625            "Selection init",
626        );
627
628        Ok(XWmSelection {
629            atom,
630            type_: selection,
631            conn: conn.clone(),
632            window,
633            owner: window,
634            mime_types: Vec::new(),
635            timestamp: x11rb::CURRENT_TIME,
636            incoming: Vec::new(),
637            outgoing: Vec::new(),
638        })
639    }
640}

929    /// Raises a window in the internal X11 state
930    ///
931    /// Needs to be called to match raising of windows inside the compositor to keep the stacking order
932    /// in sync with the compositor to avoid erroneous behavior.
933    pub fn raise_window<'a, W: X11Relatable + 'a>(&mut self, window: &'a W) -> Result<(), ConnectionError> {
934        if let Some(elem) = self.windows.iter().find(|s| window.is_window(s)) {
935            if self.client_list_stacking.last() == Some(&elem.window_id()) {
936                return Ok(());
937            }
938
939            let _guard = scopeguard::guard((), |_| {
940                let _ = self.conn.ungrab_server();
941                let _ = self.conn.flush();
942            });
943            self.conn.grab_server()?;
944            self.conn.configure_window(
945                elem.mapped_window_id().unwrap_or_else(|| elem.window_id()),
946                &ConfigureWindowAux::new().stack_mode(StackMode::ABOVE),
947            )?;
948            self.client_list_stacking.retain(|e| *e != elem.window_id());
949            self.client_list_stacking.push(elem.window_id());
950            self.conn.change_property32(
951                PropMode::REPLACE,
952                self.screen.root,
953                self.atoms._NET_CLIENT_LIST_STACKING,
954                AtomEnum::WINDOW,
955                &self.client_list_stacking,
956            )?;
957        }
958        Ok(())
959    }

961    fn update_stacking_order_impl<'a, W: X11Relatable + 'a>(
962        &mut self,
963        order: impl Iterator<Item = &'a W>,
964        direction: StackingDirection,
965    ) -> Result<(), ConnectionError> {
966        let _guard = scopeguard::guard((), |_| {
967            let _ = self.conn.ungrab_server();
968            let _ = self.conn.flush();
969        });
970        self.conn.grab_server()?;
971
972        let mut last_pos = None;
973        let mut changed = false;
974
975        for relatable in order {
976            let stacking_ordered_elems: Vec<&X11Surface> = self
977                .client_list_stacking
978                .iter()
979                .filter_map(|w| self.windows.iter().find(|s| s.window_id() == *w))
980                .collect();
981            let pos = stacking_ordered_elems.iter().position(|w| relatable.is_window(w));
982            if let (Some(pos), Some(last_pos)) = (pos, last_pos) {
983                if direction.pos_comparator(pos, last_pos) {
984                    let sibling = stacking_ordered_elems[last_pos];
985                    let sibling_id = self.client_list_stacking[last_pos];
986                    let elem = stacking_ordered_elems[pos];
987                    let elem_id = self.client_list_stacking.remove(pos);
988                    self.conn.configure_window(
989                        elem.mapped_window_id().unwrap_or(elem_id),
990                        &ConfigureWindowAux::new()
991                            .sibling(sibling.mapped_window_id().unwrap_or(sibling_id))
992                            .stack_mode(direction.stack_mode()),
993                    )?;
994                    self.client_list_stacking.insert(last_pos, elem_id);
995                    changed = true;
996                    continue;
997                }
998            }
999
1000            if pos.is_some() {
1001                last_pos = pos;
1002            }
1003        }
1004
1005        if changed {
1006            self.conn.change_property32(
1007                PropMode::REPLACE,
1008                self.screen.root,
1009                self.atoms._NET_CLIENT_LIST_STACKING,
1010                AtomEnum::WINDOW,
1011                &self.client_list_stacking,
1012            )?;
1013        }
1014        Ok(())
1015    }

1017    /// Updates the stacking order by matching provided windows downwards.
1018    ///
1019    /// This function reorders provided x11 windows in such a way,
1020    /// that windows inside the internal X11 stack follow the provided `order`
1021    /// without moving other windows around as much as possible. The internal stack
1022    /// stores windows bottom -> top, so order here is also bottom -> top.
1023    ///
1024    /// Window IDs unknown to this XWM will be ignored.
1025    /// The first window in `order` found will not be moved.
1026    ///
1027    /// If a window is encountered in `order` that is stacked below the first window,
1028    /// it will be moved to be directly above the previous window in `order`.
1029    /// E.g. Windows `C -> A -> B -> E` given in order with an internal stack of `D -> A -> B -> C`,
1030    /// will be reordered as `D -> C -> A -> B`. First `A` is moved to be directly
1031    /// above `C`, then `B`is moved to be directly above `A`.
1032    ///
1033    /// Windows in the internal stack, that are not present in `order`
1034    /// will be skipped over in the process.
1035    ///
1036    /// So if windows `A -> C` are given in order and the internal stack is `A -> B -> C`,
1037    /// no reordering will occur.
1038    ///
1039    /// See [`X11Wm::update_stacking_order_upwards`] for a variant of this algorithm,
1040    /// which works from the bottom up or [`X11Wm::raise_window`] for an easier but
1041    /// much more limited way to reorder.
1042    pub fn update_stacking_order_downwards<'a, W: X11Relatable + 'a>(
1043        &mut self,
1044        order: impl Iterator<Item = &'a W>,
1045    ) -> Result<(), ConnectionError> {
1046        self.update_stacking_order_impl(order, StackingDirection::Downwards)
1047    }

1050    ///
1051    /// This function reorders provided x11 windows in such a way,
1052    /// that windows inside the internal X11 stack follow the provided `order`
1053    /// in reverse without moving other windows around as much as possible. The
1054    /// internal stack stores windows bottom -> top, so due to the reversal, order
1055    /// here is top -> bottom.
1056    ///
1057    /// Window IDs unknown to this XWM will be ignored.
1058    /// The first window in `order` found will not be moved.
1059    ///
1060    /// If a window is encountered in `order` that is stacked above the first window,
1061    /// it will be moved to be directly below the previous window in `order`.
1062    /// E.g. Windows C -> A -> B given in order with an internal stack of `D -> A -> B -> C`,
1063    /// will be reordered as `D -> B -> A -> C`. `A` is below `C`, so it isn't moved,
1064    /// then `B` is moved to be directly below `A`.
1065    ///
1066    /// Windows in the internal stack, that are not present in `order`
1067    /// will be skipped over in the process.
1068    ///
1069    /// So if windows `A -> C` are given in order and the internal stack is `C -> B -> A`,
1070    /// no reordering will occur.
1071    ///  
1072    /// See [`X11Wm::update_stacking_order_downwards`] for a variant of this algorithm,
1073    /// which works from the top down or [`X11Wm::raise_window`] for an easier but
1074    /// much more limited way to reorder.
1075    pub fn update_stacking_order_upwards<'a, W: X11Relatable + 'a>(
1076        &mut self,
1077        order: impl Iterator<Item = &'a W>,
1078    ) -> Result<(), ConnectionError> {
1079        self.update_stacking_order_impl(order, StackingDirection::Upwards)

1165    /// Request to transfer the active `selection` for the provided `mime_type` to the provided file descriptor.
1166    pub fn send_selection<D>(
1167        &mut self,
1168        selection: SelectionTarget,
1169        mime_type: String,
1170        fd: OwnedFd,
1171        loop_handle: LoopHandle<'_, D>,
1172    ) -> Result<(), SelectionError>
1173    where
1174        D: XwmHandler + 'static,
1175    {
1176        let xwm_id = self.id();
1177        let selection = match selection {
1178            SelectionTarget::Clipboard => &mut self.clipboard,
1179            SelectionTarget::Primary => &mut self.primary,
1180        };
1181
1182        info!(
1183            selection = ?selection.type_,
1184            ?mime_type,
1185            "Send request from XWayland",
1186        );
1187
1188        let atom = match &*mime_type {
1189            "text/plain;charset=utf-8" => self.atoms.UTF8_STRING,
1190            "text/plain" => self.atoms.TEXT,
1191            x => {
1192                let prop = self
1193                    .conn
1194                    .get_property(true, selection.window, self.atoms.TARGETS, AtomEnum::ANY, 0, 4096)?
1195                    .reply_unchecked()?
1196                    .ok_or(SelectionError::UnableToDetermineAtom)?;
1197                if prop.type_ != AtomEnum::ATOM.into() {
1198                    return Err(SelectionError::UnableToDetermineAtom);
1199                }
1200                let values = prop.value32().ok_or(SelectionError::UnableToDetermineAtom)?;
1201
1202                let Some(atom) = values
1203                    .filter_map(|atom| {
1204                        let cookie = self.conn.get_atom_name(atom).ok()?;
1205                        let reply = cookie.reply_unchecked().ok()?;
1206                        std::str::from_utf8(&reply?.name)
1207                            .ok()
1208                            .map(|name| (atom, name.to_string()))
1209                    })
1210                    .find_map(|(atom, name)| if name == x { Some(atom) } else { None })
1211                else {
1212                    return Err(SelectionError::UnableToDetermineAtom);
1213                };
1214
1215                atom
1216            }
1217        };
1218
1219        debug!("Mime-type {:?} / Atom {:?}", mime_type, atom);
1220
1221        let incoming_window = self.conn.generate_id()?;
1222        self.conn.create_window(
1223            self.screen.root_depth,
1224            incoming_window,
1225            self.screen.root,
1226            0,
1227            0,
1228            1,
1229            1,
1230            0,
1231            WindowClass::INPUT_OUTPUT,
1232            self.screen.root_visual,
1233            &CreateWindowAux::new().event_mask(EventMask::PROPERTY_CHANGE),
1234        )?;
1235        self.conn.convert_selection(
1236            incoming_window,
1237            selection.atom,
1238            atom,
1239            self.atoms._WL_SELECTION,
1240            x11rb::CURRENT_TIME,
1241        )?;
1242
1243        if let Err(err) = rustix::fs::fcntl_setfl(&fd, OFlags::WRONLY | OFlags::NONBLOCK) {
1244            warn!(?err, "Failed to restrict wl file descriptor");
1245        }
1246
1247        let selection_type = selection.type_;
1248        let loop_handle_clone = loop_handle.clone();
1249        let token = loop_handle
1250            .insert_source(
1251                Generic::new(fd, Interest::WRITE, Mode::Level),
1252                move |_, fd, data| {
1253                    let xwm = data.xwm_state(xwm_id);
1254                    let conn = &xwm.conn;
1255                    let atoms = &xwm.atoms;
1256                    let selection = match selection_type {
1257                        SelectionTarget::Clipboard => &mut xwm.clipboard,
1258                        SelectionTarget::Primary => &mut xwm.primary,
1259                    };
1260                    if let Some(transfer) = selection
1261                        .incoming
1262                        .iter_mut()
1263                        .find(|t| t.window == incoming_window)
1264                    {
1265                        match write_selection_callback(fd.as_fd(), conn, atoms, transfer) {
1266                            Ok(IncomingAction::WaitForWritable) => return Ok(PostAction::Continue),
1267                            Ok(IncomingAction::WaitForProperty) if !transfer.incr_done => {
1268                                return Ok(PostAction::Disable)
1269                            }
1270                            Ok(_) | Err(_) => {
1271                                if let Some(pos) = selection
1272                                    .incoming
1273                                    .iter()
1274                                    .position(|t| t.window == incoming_window)
1275                                {
1276                                    selection.incoming.remove(pos).destroy(&loop_handle_clone);
1277                                }
1278                            }
1279                        };
1280                    }
1281                    Ok(PostAction::Remove)
1282                },
1283            )
1284            .map_err(|err| err.error)?;
1285        loop_handle.disable(&token)?;
1286
1287        let transfer = IncomingTransfer {
1288            conn: self.conn.clone(),
1289            token: Some(token),
1290            window: incoming_window,
1291            incr: false,
1292            source_data: Vec::new(),
1293            incr_done: false,
1294        };
1295        selection.incoming.push(transfer);
1296
1297        self.conn.flush()?;
1298        Ok(())
1299    }

1317    /// Gets the current primary output as advertised by xrandr
1318    pub fn get_randr_primary_output(&self) -> Result<Option<String>, ReplyError> {

1320            .conn
1321            .randr_get_output_primary(self.screen.root)?
1322            .reply()?
1323            .output;
1324        if current_primary == x11rb::NONE {
1325            return Ok(None);
1326        }
1327
1328        let res = self
1329            .conn
1330            .randr_get_screen_resources_current(self.screen.root)?

1333            .conn

1335            .reply()?;

1337            std::str::from_utf8(&info.name)

1343    /// Updates the primary output as advertised by xrandr
1344    pub fn set_randr_primary_output(&mut self, output: Option<&Output>) -> Result<(), PrimaryOutputError> {
1345        let current_primary = self
1346            .conn
1347            .randr_get_output_primary(self.screen.root)?
1348            .reply()?
1349            .output;
1350        let Some(output) = output else {
1351            if current_primary != x11rb::NONE {
1352                let cookie = self
1353                    .conn
1354                    .randr_set_output_primary(self.screen.root, x11rb::NONE)?;
1355                self.sequences_to_ignore
1356                    .push(Reverse(cookie.sequence_number() as u16));
1357            }
1358            return Ok(());
1359        };
1360
1361        let res = self
1362            .conn
1363            .randr_get_screen_resources_current(self.screen.root)?
1364            .reply()?;
1365        for output_xid in res.outputs {
1366            let info = self
1367                .conn
1368                .randr_get_output_info(output_xid, res.config_timestamp)?
1369                .reply()?;
1370            let name =
1371                std::str::from_utf8(&info.name).expect("X11 protocol violation, output name not utf-8");
1372            if name == output.name() {
1373                // we got our output
1374                if output_xid != current_primary {
1375                    let cookie = self.conn.randr_set_output_primary(self.screen.root, output_xid)?;
1376                    self.sequences_to_ignore
1377                        .push(Reverse(cookie.sequence_number() as u16));
1378                    return Ok(());
1379                }
1380            }
1381        }
1382
1383        Err(PrimaryOutputError::OutputUnknown)
1384    }

1386
1387fn handle_event<D>(
1388    loop_handle: &LoopHandle<'_, D>,
1389    state: &mut D,
1390    xwm_id: XwmId,
1391    event: Event,
1392) -> Result<(), ReplyOrIdError>
1393where
1394    D: XwmHandler,
1395    D: xwayland_shell::XWaylandShellHandler,
1396    D: 'static,
1397{
1398    let xwm = state.xwm_state(xwm_id);
1399    let _guard = xwm.span.enter();
1400    let conn = xwm.conn.clone();
1401
1402    let mut should_ignore = false;
1403    if let Some(seqno) = event.wire_sequence_number() {
1404        // Check sequences_to_ignore and remove entries with old (=smaller) numbers.
1405        while let Some(&Reverse(to_ignore)) = xwm.sequences_to_ignore.peek() {
1406            // Sequence numbers can wrap around, so we cannot simply check for
1407            // "to_ignore <= seqno". This is equivalent to "to_ignore - seqno <= 0", which is what we
1408            // check instead. Since sequence numbers are unsigned, we need a trick: We decide
1409            // that values from [MAX/2, MAX] count as "<= 0" and the rest doesn't.
1410            if to_ignore.wrapping_sub(seqno) <= u16::MAX / 2 {
1411                // If the two sequence numbers are equal, this event should be ignored.
1412                should_ignore = to_ignore == seqno;
1413                break;
1414            }
1415            xwm.sequences_to_ignore.pop();
1416        }
1417    }
1418
1419    debug!(
1420        event = ?event,
1421        should_ignore = should_ignore,
1422        "Got X11 event",
1423    );
1424    if should_ignore {
1425        return Ok(());
1426    }
1427
1428    match event {
1429        Event::CreateNotify(n) => {
1430            if n.window == xwm.wm_window
1431                || n.window == xwm.xsettings.window
1432                || n.window == xwm.clipboard.window
1433                || xwm.clipboard.incoming.iter().any(|i| n.window == i.window)
1434                || n.window == xwm.primary.window
1435                || xwm.primary.incoming.iter().any(|i| n.window == i.window)
1436            {
1437                return Ok(());
1438            }
1439
1440            xwm.conn.change_window_attributes(
1441                n.window,
1442                &ChangeWindowAttributesAux::new().event_mask(EventMask::PROPERTY_CHANGE),
1443            )?;
1444            xwm.conn.flush()?;
1445
1446            if xwm.windows.iter().any(|s| s.mapped_window_id() == Some(n.window)) {
1447                return Ok(());
1448            }
1449
1450            let geo = conn.get_geometry(n.window)?.reply()?;
1451            let geometry = Rectangle::<i32, Client>::new(
1452                (geo.x as i32, geo.y as i32).into(),
1453                (geo.width as i32, geo.height as i32).into(),
1454            )
1455            .to_logical(xwm.client_scale.load(Ordering::Acquire) as i32);
1456
1457            let surface = X11Surface::new(
1458                Some(xwm),
1459                n.window,
1460                n.override_redirect,
1461                Arc::downgrade(&conn),
1462                xwm.atoms,
1463                geometry,
1464            );
1465            surface.update_properties()?;
1466            xwm.windows.push(surface.clone());
1467
1468            drop(_guard);
1469            if n.override_redirect {
1470                state.new_override_redirect_window(xwm_id, surface);
1471            } else {
1472                state.new_window(xwm_id, surface);
1473            }
1474        }
1475        Event::MapRequest(r) => {
1476            if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == r.window).cloned() {
1477                if surface.state.lock().unwrap().mapped_onto.is_none() {
1478                    // we reparent windows, because a lot of stuff expects, that we do
1479                    let geo = conn.get_geometry(r.window)?.reply()?;
1480                    let win = r.window;
1481                    let frame_win = conn.generate_id()?;
1482                    let win_aux = CreateWindowAux::new().event_mask(
1483                        EventMask::SUBSTRUCTURE_NOTIFY
1484                            | EventMask::SUBSTRUCTURE_REDIRECT
1485                            | EventMask::PROPERTY_CHANGE,
1486                    );
1487
1488                    {
1489                        let _guard = scopeguard::guard((), |_| {
1490                            let _ = conn.ungrab_server();
1491                        });
1492
1493                        conn.grab_server()?;
1494                        let cookie1 = conn.create_window(
1495                            COPY_DEPTH_FROM_PARENT,
1496                            frame_win,
1497                            xwm.screen.root,
1498                            geo.x,
1499                            geo.y,
1500                            geo.width,
1501                            geo.height,
1502                            0,
1503                            WindowClass::INPUT_OUTPUT,
1504                            x11rb::COPY_FROM_PARENT,
1505                            &win_aux,
1506                        )?;
1507                        let cookie2 = conn.reparent_window(win, frame_win, 0, 0)?;
1508                        conn.map_window(win)?;
1509
1510                        // Ignore all events caused by reparent_window(). All those events have the sequence number
1511                        // of the reparent_window() request, thus remember its sequence number. The
1512                        // grab_server()/ungrab_server() is done so that the server does not handle other clients
1513                        // in-between, which could cause other events to get the same sequence number.
1514                        xwm.sequences_to_ignore
1515                            .push(Reverse(cookie1.sequence_number() as u16));
1516                        xwm.sequences_to_ignore
1517                            .push(Reverse(cookie2.sequence_number() as u16));
1518                    }
1519
1520                    surface.state.lock().unwrap().mapped_onto = Some(frame_win);
1521
1522                    // A MapRequest can only happen when an X11 window does not have the
1523                    // override-redirect flag set. It's possible for a window to be created as
1524                    // override-redirect and later set the flag to false before mapping.
1525                    // In that case, we set the X11Surface's override-redirect state to false here
1526                    // to prevent `set_mapped` and `configure` from failing.
1527                    surface.state.lock().unwrap().override_redirect = false;
1528
1529                    drop(_guard);
1530                    state.map_window_request(xwm_id, surface);
1531                }
1532            }
1533        }
1534        Event::MapNotify(n) => {
1535            trace!(window = ?n.window, "mapped X11 Window");
1536            if let Some(surface) = xwm
1537                .windows
1538                .iter()
1539                .find(|x|
1540                      // don't include the reparenting windows
1541                      (x.window_id() == n.window && n.override_redirect) ||
1542                      x.mapped_window_id() == Some(n.window))
1543                .cloned()
1544            {
1545                // Client may have changed override-redirect flag since window creation
1546                surface.state.lock().unwrap().override_redirect = n.override_redirect;
1547
1548                if n.override_redirect {
1549                    drop(_guard);
1550                    state.mapped_override_redirect_window(xwm_id, surface);
1551                } else {
1552                    xwm.client_list.push(surface.window_id());
1553                    xwm.client_list_stacking.push(surface.window_id());
1554                    conn.change_property32(
1555                        PropMode::APPEND,
1556                        xwm.screen.root,
1557                        xwm.atoms._NET_CLIENT_LIST,
1558                        AtomEnum::WINDOW,
1559                        &[surface.window_id()],
1560                    )?;
1561                    conn.change_property32(
1562                        PropMode::APPEND,
1563                        xwm.screen.root,
1564                        xwm.atoms._NET_CLIENT_LIST_STACKING,
1565                        AtomEnum::WINDOW,
1566                        &[surface.window_id()],
1567                    )?;
1568                    drop(_guard);
1569                    state.map_window_notify(xwm_id, surface);
1570                }
1571            }
1572        }
1573        Event::ConfigureRequest(r) => {
1574            if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == r.window).cloned() {
1575                drop(_guard);
1576
1577                let client_scale = xwm.client_scale.load(Ordering::Acquire);
1578
1579                // Pass the request to downstream to decide
1580                state.configure_request(
1581                    xwm_id,
1582                    surface.clone(),
1583                    if u16::from(r.value_mask) & u16::from(ConfigWindow::X) != 0 {
1584                        Some(((r.x as f64) / client_scale).round() as i32)
1585                    } else {
1586                        None
1587                    },
1588                    if u16::from(r.value_mask) & u16::from(ConfigWindow::Y) != 0 {
1589                        Some(((r.y as f64) / client_scale).round() as i32)
1590                    } else {
1591                        None
1592                    },
1593                    if u16::from(r.value_mask) & u16::from(ConfigWindow::WIDTH) != 0 {
1594                        Some(((r.width as f64) / client_scale).round() as u32)
1595                    } else {
1596                        None
1597                    },
1598                    if u16::from(r.value_mask) & u16::from(ConfigWindow::HEIGHT) != 0 {
1599                        Some(((r.height as f64) / client_scale).round() as u32)
1600                    } else {
1601                        None
1602                    },
1603                    if u16::from(r.value_mask) & u16::from(ConfigWindow::STACK_MODE) != 0 {
1604                        match r.stack_mode {
1605                            StackMode::ABOVE => {
1606                                if u16::from(r.value_mask) & u16::from(ConfigWindow::SIBLING) != 0 {
1607                                    Some(Reorder::Above(r.sibling))
1608                                } else {
1609                                    Some(Reorder::Top)
1610                                }
1611                            }
1612                            StackMode::BELOW => {
1613                                if u16::from(r.value_mask) & u16::from(ConfigWindow::SIBLING) != 0 {
1614                                    Some(Reorder::Below(r.sibling))
1615                                } else {
1616                                    Some(Reorder::Bottom)
1617                                }
1618                            }
1619                            _ => None,
1620                        }
1621                    } else {
1622                        None
1623                    },
1624                );
1625                // Synthetic event
1626                surface.configure(None).map_err(|err| match err {
1627                    X11SurfaceError::Connection(err) => err,
1628                    X11SurfaceError::UnsupportedForOverrideRedirect => unreachable!(),
1629                })?;
1630            }
1631        }
1632        Event::ConfigureNotify(n) => {
1633            trace!(window = ?n, "configured X11 Window");
1634
1635            let client_scale = xwm.client_scale.load(Ordering::Acquire);
1636            let geometry = Rectangle::<i32, Client>::new(
1637                (n.x as i32, n.y as i32).into(),
1638                (n.width as i32, n.height as i32).into(),
1639            )
1640            .to_f64()
1641            .to_logical(client_scale)
1642            .to_i32_round();
1643
1644            if let Some(surface) = xwm
1645                .windows
1646                .iter()
1647                .find(|x| x.mapped_window_id() == Some(n.window))
1648                .cloned()
1649            {
1650                drop(_guard);
1651                state.configure_notify(
1652                    xwm_id,
1653                    surface,
1654                    geometry,
1655                    if n.above_sibling == x11rb::NONE {
1656                        None
1657                    } else {
1658                        Some(n.above_sibling)
1659                    },
1660                );
1661            } else if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == n.window).cloned() {
1662                if surface.is_override_redirect() {
1663                    surface.state.lock().unwrap().geometry = geometry;
1664                    drop(_guard);
1665                    state.configure_notify(
1666                        xwm_id,
1667                        surface,
1668                        geometry,
1669                        if n.above_sibling == x11rb::NONE {
1670                            None
1671                        } else {
1672                            Some(n.above_sibling)
1673                        },
1674                    );
1675                }
1676            }
1677        }
1678        Event::UnmapNotify(n) => {
1679            if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == n.window).cloned() {
1680                xwm.client_list.retain(|w| *w != surface.window_id());
1681                xwm.client_list_stacking.retain(|w| *w != surface.window_id());
1682                {
1683                    let _guard = scopeguard::guard((), |_| {
1684                        let _ = conn.ungrab_server();
1685                    });
1686                    conn.grab_server()?;
1687                    conn.change_property32(
1688                        PropMode::REPLACE,
1689                        xwm.screen.root,
1690                        xwm.atoms._NET_CLIENT_LIST,
1691                        AtomEnum::WINDOW,
1692                        &xwm.client_list,
1693                    )?;
1694                    conn.change_property32(
1695                        PropMode::REPLACE,
1696                        xwm.screen.root,
1697                        xwm.atoms._NET_CLIENT_LIST_STACKING,
1698                        AtomEnum::WINDOW,
1699                        &xwm.client_list_stacking,
1700                    )?;
1701                    {
1702                        let mut state = surface.state.lock().unwrap();
1703                        conn.reparent_window(
1704                            n.window,
1705                            xwm.screen.root,
1706                            state.geometry.loc.x as i16,
1707                            state.geometry.loc.y as i16,
1708                        )?;
1709                        if let Some(frame) = state.mapped_onto.take() {
1710                            conn.destroy_window(frame)?;
1711                        }
1712                    }
1713                }
1714                drop(_guard);
1715                state.unmapped_window(xwm_id, surface.clone());
1716                {
1717                    let mut state = surface.state.lock().unwrap();
1718                    state.wl_surface = None;
1719                }
1720            }
1721        }
1722        Event::DestroyNotify(n) => {
1723            if let Some(selection) = if xwm.clipboard.incoming.iter().any(|t| t.window == n.window)
1724                || xwm
1725                    .clipboard
1726                    .outgoing
1727                    .iter()
1728                    .any(|t| t.request.requestor == n.window)
1729            {
1730                Some(&mut xwm.clipboard)
1731            } else if xwm.primary.incoming.iter().any(|t| t.window == n.window)
1732                || xwm
1733                    .primary
1734                    .outgoing
1735                    .iter()
1736                    .any(|t| t.request.requestor == n.window)
1737            {
1738                Some(&mut xwm.primary)
1739            } else {
1740                None
1741            } {
1742                // TODO: drain_filter
1743                let mut i = 0;
1744                while i < selection.incoming.len() {
1745                    if selection.incoming[i].window == n.window {
1746                        selection.incoming.remove(i).destroy(loop_handle);
1747                    } else {
1748                        i += 1;
1749                    }
1750                }
1751                let mut i = 0;
1752                while i < selection.outgoing.len() {
1753                    if selection.outgoing[i].request.requestor == n.window {
1754                        selection.outgoing.remove(i).destroy(loop_handle);
1755                    } else {
1756                        i += 1;
1757                    }
1758                }
1759            }
1760
1761            if let Some(pos) = xwm.windows.iter().position(|x| x.window_id() == n.window) {
1762                let surface = xwm.windows.remove(pos);
1763                surface.state.lock().unwrap().alive = false;
1764                drop(_guard);
1765                state.destroyed_window(xwm_id, surface);
1766            }
1767        }
1768        Event::XfixesSelectionNotify(n) => {
1769            let selection = match n.selection {
1770                x if x == xwm.atoms.CLIPBOARD => &mut xwm.clipboard,
1771                x if x == xwm.atoms.PRIMARY => &mut xwm.primary,
1772                _ => return Ok(()),
1773            };
1774
1775            selection.owner = n.owner;
1776            if selection.owner == selection.window {
1777                selection.timestamp = n.timestamp;
1778                return Ok(());
1779            }
1780
1781            if n.owner == x11rb::NONE && selection.owner != selection.window {
1782                // A real X clients selection went away, not our proxy
1783                let selection = selection.type_;
1784                drop(_guard);
1785                state.cleared_selection(xwm_id, selection);
1786                return Ok(());
1787            }
1788
1789            // Actually query the new selection, which will give us a SelectioNotify event
1790            conn.convert_selection(
1791                selection.window,
1792                selection.atom,
1793                xwm.atoms.TARGETS,
1794                xwm.atoms._WL_SELECTION,
1795                n.timestamp,
1796            )?;
1797        }
1798        Event::SelectionNotify(n) => {
1799            let selection = match n.selection {
1800                x if x == xwm.atoms.CLIPBOARD => &mut xwm.clipboard,
1801                x if x == xwm.atoms.PRIMARY => &mut xwm.primary,
1802                _ => return Ok(()),
1803            };
1804
1805            match n.target {
1806                x if x == xwm.atoms.TARGETS => {
1807                    if let Some(prop) = conn
1808                        .get_property(
1809                            true,
1810                            selection.window,
1811                            xwm.atoms._WL_SELECTION,
1812                            AtomEnum::ANY,
1813                            0,
1814                            4096,
1815                        )?
1816                        .reply_unchecked()?
1817                    {
1818                        if prop.type_ == AtomEnum::ATOM.into() {
1819                            if let Some(values) = prop.value32() {
1820                                let mime_types = values
1821                                    .filter_map(|val| {
1822                                        match val {
1823                                            val if val == xwm.atoms.UTF8_STRING => {
1824                                                Some(Ok(String::from("text/plain;charset=utf-8")))
1825                                            }
1826                                            val if val == xwm.atoms.TEXT => {
1827                                                Some(Ok(String::from("text/plain")))
1828                                            }
1829                                            val if val == xwm.atoms.TARGETS || val == xwm.atoms.TIMESTAMP => {
1830                                                None
1831                                            }
1832                                            val => {
1833                                                let cookie = match conn.get_atom_name(val) {
1834                                                    Ok(cookie) => cookie,
1835                                                    Err(err) => return Some(Err(err)),
1836                                                };
1837                                                let reply = match cookie.reply_unchecked() {
1838                                                    Ok(reply) => reply,
1839                                                    Err(err) => return Some(Err(err)),
1840                                                };
1841                                                if let Some(reply) = reply {
1842                                                    if let Ok(name) = std::str::from_utf8(&reply.name) {
1843                                                        if name.contains('/') {
1844                                                            // hopefully a mime-type
1845                                                            Some(Ok(String::from(name)))
1846                                                        } else {
1847                                                            None
1848                                                        }
1849                                                    } else {
1850                                                        None
1851                                                    }
1852                                                } else {
1853                                                    None
1854                                                }
1855                                            }
1856                                        }
1857                                    })
1858                                    .collect::<Result<Vec<String>, _>>()?;
1859
1860                                let selection = selection.type_;
1861                                drop(_guard);
1862                                state.new_selection(xwm_id, selection, mime_types);
1863                            }
1864                        }
1865                    }
1866                }
1867                x if x == AtomEnum::NONE.into() => {
1868                    // transfer failed
1869                    if let Some(pos) = selection.incoming.iter().position(|t| t.window == n.requestor) {
1870                        selection.incoming.remove(pos).destroy(loop_handle);
1871                    }
1872                }
1873                _ => {
1874                    let Some(transfer) = selection.incoming.iter_mut().find(|t| t.window == n.requestor)
1875                    else {
1876                        return Ok(());
1877                    };
1878
1879                    if let Some(prop) = conn
1880                        .get_property(
1881                            true,
1882                            transfer.window,
1883                            xwm.atoms._WL_SELECTION,
1884                            AtomEnum::ANY,
1885                            0,
1886                            0x1fffffff,
1887                        )?
1888                        .reply_unchecked()?
1889                    {
1890                        let type_ = prop.type_;
1891                        transfer.read_selection_prop(prop);
1892                        if type_ == xwm.atoms.INCR {
1893                            transfer.incr = true;
1894                            return Ok(());
1895                        } else if let Some(token) = transfer.token.as_ref() {
1896                            let _ = loop_handle.enable(token);
1897                        } else if let Some(pos) =
1898                            selection.incoming.iter().position(|t| t.window == n.requestor)
1899                        {
1900                            selection.incoming.remove(pos);
1901                        }
1902                    }
1903                }
1904            }
1905        }
1906        Event::SelectionRequest(n) => {
1907            let selection_type = match n.selection {
1908                x if x == xwm.atoms.CLIPBOARD => xwm.clipboard.type_,
1909                x if x == xwm.atoms.PRIMARY => xwm.primary.type_,
1910                _ => {
1911                    warn!(
1912                        target = ?n.selection,
1913                        "Got SelectionRequest for unknown Target",
1914                    );
1915                    send_selection_notify_resp(&conn, &n, false)?;
1916                    return Ok(());
1917                }
1918            };
1919
1920            // work around borrowing
1921            drop(_guard);

2470fn read_selection_callback(
2471    conn: &RustConnection,
2472    atoms: &Atoms,
2473    fd: BorrowedFd<'_>,
2474    transfer: &mut OutgoingTransfer,
2475) -> Result<OutgoingAction, ReplyOrIdError> {
2476    let mut buf = [0; INCR_CHUNK_SIZE];
2477    let Ok(len) = rustix::io::read(fd, &mut buf) else {
2478        debug!(
2479            requestor = transfer.request.requestor,
2480            "File descriptor closed, aborting transfer."
2481        );
2482        send_selection_notify_resp(conn, &transfer.request, false)?;
2483        return Ok(OutgoingAction::Done);
2484    };
2485    trace!(
2486        requestor = transfer.request.requestor,
2487        "Transfer became readable, read {} bytes",
2488        len
2489    );
2490
2491    transfer.source_data.extend_from_slice(&buf[..len]);
2492    if transfer.source_data.len() >= INCR_CHUNK_SIZE {
2493        if !transfer.incr {
2494            // start incr transfer
2495            trace!(
2496                requestor = transfer.request.requestor,
2497                "Transfer became incremental",
2498            );
2499            conn.change_property32(
2500                PropMode::REPLACE,
2501                transfer.request.requestor,
2502                transfer.request.property,
2503                atoms.INCR,
2504                &[INCR_CHUNK_SIZE as u32],
2505            )?;
2506            conn.flush()?;
2507            transfer.incr = true;
2508            transfer.property_set = true;
2509            transfer.flush_property_on_delete = true;
2510            send_selection_notify_resp(conn, &transfer.request, true)?;
2511        } else if transfer.property_set {
2512            // got more bytes, waiting for property delete
2513            transfer.flush_property_on_delete = true;
2514        } else {
2515            // got more bytes, property deleted
2516            let len = transfer.flush_data()?;
2517            trace!(
2518                requestor = transfer.request.requestor,
2519                "Send data chunk: {} bytes",
2520                len
2521            );
2522        }
2523    }
2524
2525    if len == 0 {
2526        if transfer.incr {
2527            debug!("Incr transfer completed");
2528            if !transfer.property_set {
2529                let len = transfer.flush_data()?;
2530                trace!(
2531                    requestor = transfer.request.requestor,
2532                    "Send data chunk: {} bytes",
2533                    len
2534                );
2535            }
2536            transfer.flush_property_on_delete = true;
2537            Ok(OutgoingAction::DoneReading)
2538        } else {
2539            let len = transfer.flush_data()?;
2540            debug!("Non-Incr transfer completed with {} bytes", len);
2541            send_selection_notify_resp(conn, &transfer.request, true)?;
2542            Ok(OutgoingAction::Done)
2543        }
2544    } else {
2545        Ok(OutgoingAction::WaitForReadable)
2546    } // nothing to be done, buffered the bytes
2547}

2563            if transfer.incr {
2564                conn.delete_property(transfer.window, atoms._WL_SELECTION)?;
2565                Ok(IncomingAction::WaitForProperty)
2566            } else {
2567                debug!(?transfer, "Non-Incr Transfer complete!");
2568                Ok(IncomingAction::Done)
2569            }
2570        }
2571        Ok(false) => Ok(IncomingAction::WaitForWritable),
2572        Err(err) => {
2573            warn!(?err, "Transfer errored");
2574            if transfer.incr {
2575                // even if it failed, we still need to drain the incr transfer
2576                conn.delete_property(transfer.window, atoms._WL_SELECTION)?;
2577            }
2578            Ok(IncomingAction::Done)
2579        }
2580    }
2581}
