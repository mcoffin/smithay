1//! Element to render a wayland surface
2//!
3//! # Why use this implementation
4//!
5//! The [`WaylandSurfaceRenderElement`] provides an easy way to
6//! integrate a [`WlSurface`](wayland_server::protocol::wl_surface::WlSurface) in the smithay rendering pipeline
7//!
8//! # How to use it
9//!
10//! [`WaylandSurfaceRenderElement::from_surface`] allows you to obtain a [`WaylandSurfaceRenderElement`] for a single [`WlSurface`](wayland_server::protocol::wl_surface::WlSurface).
11//! To retrieve [`WaylandSurfaceRenderElement`]s for a whole surface tree you can use [`render_elements_from_surface_tree`].
12//!
13//! ```no_run
14//! # #[cfg(all(
15//! #     feature = "wayland_frontend",
16//! #     feature = "backend_egl",
17//! #     feature = "use_system_lib"
18//! # ))]
19//! # use smithay::backend::{
20//! #     egl::{self, display::EGLBufferReader},
21//! #     renderer::ImportEgl,
22//! # };
23//! # use smithay::{
24//! #     backend::allocator::{Fourcc, dmabuf::Dmabuf},
25//! #     backend::renderer::{
26//! #         Color32F, DebugFlags, Frame, ImportDma, ImportDmaWl, ImportMem, ImportMemWl, Renderer, Texture,
27//! #         TextureFilter, sync::SyncPoint, test::{DummyRenderer, DummyFramebuffer},
28//! #     },
29//! #     utils::{Buffer, Physical},
30//! #     wayland::compositor::SurfaceData,
31//! # };
32//! use smithay::{
33//!     backend::renderer::{
34//!         damage::OutputDamageTracker,
35//!         element::{
36//!             Kind,
37//!             surface::{render_elements_from_surface_tree, WaylandSurfaceRenderElement},
38//!         },
39//!     },
40//!     utils::{Point, Rectangle, Size, Transform},
41//! };
42//! # use wayland_server::{backend::ObjectId, protocol::wl_surface::WlSurface, Resource};
43//! # let display = wayland_server::Display::<()>::new().unwrap();
44//! # let dh = display.handle();
45//! # let surface = WlSurface::from_id(&dh, ObjectId::null()).unwrap();
46//!
47//! // Initialize a static damage tracked renderer
48//! let mut damage_tracker = OutputDamageTracker::new((800, 600), 1.0, Transform::Normal);
49//! # let mut renderer = DummyRenderer::default();
50//! # let mut framebuffer = DummyFramebuffer;
51//!
52//! loop {
53//!     // Create the render elements from the surface
54//!     let location = Point::from((100, 100));
55//!     let render_elements: Vec<WaylandSurfaceRenderElement<DummyRenderer>> =
56//!         render_elements_from_surface_tree(&mut renderer, &surface, location, 1.0, 1.0, Kind::Unspecified);
57//!
58//!     // Render the element(s)
59//!     damage_tracker
60//!         .render_output(&mut renderer, &mut framebuffer, 0, &*render_elements, [0.8, 0.8, 0.9, 1.0])
61//!         .expect("failed to render output");
62//! }
63//! ```

87/// Retrieve the [`WaylandSurfaceRenderElement`]s for a surface tree
88#[instrument(level = "trace", skip(renderer, location, scale))]
89#[profiling::function]
90pub fn render_elements_from_surface_tree<R, E>(
91    renderer: &mut R,
92    surface: &wl_surface::WlSurface,
93    location: impl Into<Point<i32, Physical>>,
94    scale: impl Into<Scale<f64>>,
95    alpha: f32,
96    kind: Kind,
97) -> Vec<E>
98where
99    R: Renderer + ImportAll,
100    R::TextureId: Clone + 'static,
101    E: From<WaylandSurfaceRenderElement<R>>,
102{
103    let location = location.into().to_f64();
104    let scale = scale.into();
105    let mut surfaces: Vec<E> = Vec::new();
106
107    compositor::with_surface_tree_downward(
108        surface,
109        location,
110        |_, states, location| {
111            let mut location = *location;
112            let data = states.data_map.get::<RendererSurfaceStateUserData>();
113
114            if let Some(data) = data {
115                if let Some(view) = data.lock().unwrap().view() {
116                    location += view.offset.to_f64().to_physical(scale);
117                    TraversalAction::DoChildren(location)
118                } else {
119                    TraversalAction::SkipChildren
120                }
121            } else {
122                TraversalAction::SkipChildren
123            }
124        },
125        |surface, states, location| {
126            let mut location = *location;
127            let data = states.data_map.get::<RendererSurfaceStateUserData>();
128
129            if let Some(data) = data {
130                let has_view = if let Some(view) = data.lock().unwrap().view() {
131                    location += view.offset.to_f64().to_physical(scale);
132                    true
133                } else {
134                    false
135                };
136
137                if has_view {
138                    match WaylandSurfaceRenderElement::from_surface(
139                        renderer, surface, states, location, alpha, kind,
140                    ) {
141                        Ok(Some(surface)) => surfaces.push(surface.into()),
142                        Ok(None) => {} // surface is not mapped
143                        Err(err) => {
144                            warn!("Failed to import surface: {}", err);
145                        }
146                    };
147                }
148            }
149        },
150        |_, _, _| true,
151    );
152
153    surfaces
154}

156/// Texture used for the [`WaylandSurfaceRenderElement`]
157#[derive(Debug)]
158pub enum WaylandSurfaceTexture<R: Renderer> {
159    /// A renderer texture
160    Texture(R::TextureId),
161    /// A solid color
162    SolidColor(Color32F),
163}

165/// A single surface render element
166pub struct WaylandSurfaceRenderElement<R: Renderer> {
167    id: Id,
168    location: Point<f64, Physical>,
169    alpha: f32,
170    kind: Kind,
171
172    view: SurfaceView,
173    buffer: Buffer,
174    buffer_scale: i32,
175    buffer_transform: Transform,
176    buffer_dimensions: Size<i32, BufferCoords>,
177    damage: DamageSnapshot<i32, BufferCoords>,
178    opaque_regions: OpaqueRegions<i32, Logical>,
179    texture: WaylandSurfaceTexture<R>,
180}

191impl<R: Renderer + ImportAll> WaylandSurfaceRenderElement<R> {
192    /// Create a render element from a surface
193    #[profiling::function]
194    pub fn from_surface(
195        renderer: &mut R,
196        surface: &wl_surface::WlSurface,
197        states: &SurfaceData,
198        location: Point<f64, Physical>,
199        alpha: f32,
200        kind: Kind,
201    ) -> Result<Option<Self>, R::Error>
202    where
203        R::TextureId: Clone + 'static,
204    {
205        let id = Id::from_wayland_resource(surface);
206        crate::backend::renderer::utils::import_surface(renderer, states)?;
207
208        let mut alpha_modifier_state = states.cached_state.get::<AlphaModifierSurfaceCachedState>();
209        let alpha_multiplier = alpha_modifier_state.current().multiplier_f32().unwrap_or(1.0);
210
211        let Some(data_ref) = states.data_map.get::<RendererSurfaceStateUserData>() else {
212            return Ok(None);
213        };
214        Ok(Self::from_state(
215            renderer,
216            id,
217            location,
218            alpha * alpha_multiplier,
219            kind,
220            &data_ref.lock().unwrap(),
221        ))
222    }
223
224    fn from_state(
225        renderer: &mut R,
226        id: Id,
227        location: Point<f64, Physical>,
228        alpha: f32,
229        kind: Kind,
230        data: &RendererSurfaceState,
231    ) -> Option<Self>
232    where
233        R::TextureId: Clone + 'static,
234    {
235        let buffer = data.buffer()?.clone();
236
237        let texture = if let Ok(spb) = crate::wayland::single_pixel_buffer::get_single_pixel_buffer(&buffer) {
238            WaylandSurfaceTexture::SolidColor(Color32F::from(spb.rgba32f()))
239        } else {
240            WaylandSurfaceTexture::Texture(data.texture(renderer.context_id())?.clone())
241        };
242
243        Some(Self {
244            id,
245            location,
246            alpha,
247            kind,
248            view: data.view()?,
249            buffer,
250            buffer_scale: data.buffer_scale(),
251            buffer_transform: data.buffer_transform(),
252            buffer_dimensions: data.buffer_dimensions?,
253            damage: data.damage.snapshot(),
254            opaque_regions: data
255                .opaque_regions()
256                .map(OpaqueRegions::from_slice)
257                .unwrap_or_default(),
258            texture,
259        })
260    }
261
262    fn size(&self, scale: impl Into<Scale<f64>>) -> Size<i32, Physical> {
263        ((self.view.dst.to_f64().to_physical(scale).to_point() + self.location).to_i32_round()
264            - self.location.to_i32_round())
265        .to_size()
266    }
267
268    /// Get the buffer dimensions in logical coordinates
269    pub fn buffer_size(&self) -> Size<i32, Logical> {
270        self.buffer_dimensions
271            .to_logical(self.buffer_scale, self.buffer_transform)
272    }
273
274    /// Get the view into the surface
275    pub fn view(&self) -> SurfaceView {
276        self.view
277    }
278
279    /// Get the buffer texture
280    pub fn texture(&self) -> &WaylandSurfaceTexture<R> {
281        &self.texture
282    }
283}

285impl<R: Renderer + ImportAll> Element for WaylandSurfaceRenderElement<R> {
286    fn id(&self) -> &Id {
287        &self.id
288    }
289
290    fn current_commit(&self) -> CommitCounter {
291        self.damage.current_commit()
292    }
293
294    fn geometry(&self, scale: Scale<f64>) -> Rectangle<i32, Physical> {
295        Rectangle::new(self.location.to_i32_round(), self.size(scale))
296    }
297
298    fn src(&self) -> Rectangle<f64, BufferCoords> {
299        self.view.src.to_buffer(
300            self.buffer_scale as f64,
301            self.buffer_transform,
302            &self.buffer_size().to_f64(),
303        )
304    }
305
306    fn transform(&self) -> Transform {
307        self.buffer_transform
308    }
309
310    fn damage_since(&self, scale: Scale<f64>, commit: Option<CommitCounter>) -> DamageSet<i32, Physical> {
311        let dst_size = self.size(scale);
312        self.damage
313            .damage_since(commit)
314            .unwrap_or_else(|| DamageSet::from_slice(&[Rectangle::from_size(self.buffer_dimensions)]))
315            .iter()
316            .filter_map(|rect| {
317                rect.to_f64()
318                    // first bring the damage into logical space
319                    // Note: We use f64 for this as the damage could
320                    // be not dividable by the buffer scale without
321                    // a rest
322                    .to_logical(
323                        self.buffer_scale as f64,
324                        self.buffer_transform,
325                        &self.buffer_dimensions.to_f64(),
326                    )
327                    // then crop by the surface view (viewporter for example could define a src rect)
328                    .intersection(self.view.src)
329                    // move and scale the cropped rect (viewporter could define a dst size)
330                    .map(|rect| self.view.rect_to_global(rect).to_i32_up::<i32>())
331                    // now bring the damage to physical space
332                    .map(|rect| {
333                        // We calculate the scale between to rounded
334                        // surface size and the scaled surface size
335                        // and use it to scale the damage to the rounded
336                        // surface size by multiplying the output scale
337                        // with the result.
338                        let surface_scale = dst_size.to_f64() / self.view.dst.to_f64().to_physical(scale);
339                        rect.to_physical_precise_up(surface_scale * scale)
340                    })
341            })
342            .collect::<DamageSet<_, _>>()
343    }
344
345    fn opaque_regions(&self, scale: Scale<f64>) -> OpaqueRegions<i32, Physical> {
346        if self.alpha < 1.0 {
347            return OpaqueRegions::default();
348        }
349
350        self.opaque_regions
351            .iter()
352            .map(|r| {
353                let loc = r.loc.to_physical_precise_round(scale);
354                let size = ((r.size.to_f64().to_physical(scale).to_point() + self.location).to_i32_round()
355                    - self.location.to_i32_round())
356                .to_size();
357                Rectangle::new(loc, size)
358            })
359            .collect::<OpaqueRegions<_, _>>()
360    }
361
362    fn alpha(&self) -> f32 {
363        self.alpha
364    }
365
366    fn kind(&self) -> Kind {
367        self.kind
368    }
369}

371impl<R> RenderElement<R> for WaylandSurfaceRenderElement<R>
372where
373    R: Renderer + ImportAll,
374    R::TextureId: Texture + 'static,
375{
376    #[inline]
377    fn underlying_storage(&self, _renderer: &mut R) -> Option<UnderlyingStorage<'_>> {
378        Some(UnderlyingStorage::Wayland(&self.buffer))
379    }
380
381    #[instrument(level = "trace", skip(frame))]
382    #[profiling::function]
383    fn draw(
384        &self,
385        frame: &mut R::Frame<'_, '_>,
386        src: Rectangle<f64, BufferCoords>,
387        dst: Rectangle<i32, Physical>,
388        damage: &[Rectangle<i32, Physical>],
389        opaque_regions: &[Rectangle<i32, Physical>],
390    ) -> Result<(), R::Error> {
391        match self.texture {
392            WaylandSurfaceTexture::Texture(ref texture) => frame.render_texture_from_to(
393                texture,
394                src,
395                dst,
396                damage,
397                opaque_regions,
398                self.buffer_transform,
399                self.alpha,
400            ),
401            WaylandSurfaceTexture::SolidColor(color) => frame.draw_solid(dst, damage, color * self.alpha),
402        }
403    }
