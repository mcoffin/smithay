1//! Common base for elements that can be drawn by a [`Renderer`]
2//!
3//! A [`RenderElement`] defines what should be [`draw`](RenderElement::draw)n where.
4//! Additionally it provides the foundation for effective damage tracked rendering
5//! by allowing to query for damage between two [`CommitCounter`]s.
6//!
7//! For specialized renderers it can optionally provide access to the [`UnderlyingStorage`]
8//! of the element.
9//!
10//! Out of the box smithay provides the following elements
11//! - [`memory`] - Memory based render element
12//! - [`texture`] - Texture based render element
13//! - [`surface`] - Wayland surface render element
14//! - [`solid`] - Solid color render element
15//!
16//! The [`render_elements!`] macro provides an easy way to aggregate multiple different [RenderElement]s
17//! into a single enum.
18//!
19//! See the [`damage`](crate::backend::renderer::damage) module for more information on
20//! damage tracking.

165/// Holds the primary scanout output for a surface
166#[derive(Debug, Default)]
167pub struct PrimaryScanoutOutput(Option<(WeakOutput, RenderElementState)>);
168
169impl PrimaryScanoutOutput {
170    /// Update the primary scan-out output from [`RenderElementStates`]
171    ///
172    /// If the current primary scan-out output is different than the
173    /// provided output and the element is present according to the states
174    /// the provided compare function will be run to select the preferred
175    /// output. Smithay provides a [`default`](`default_primary_scanout_output_compare`)
176    /// compare function for convenience.
177    ///
178    /// Returns the updated primary scan-out output if any
179    pub fn update_from_render_element_states<F>(
180        &mut self,
181        element_id: impl Into<Id>,
182        output: &Output,
183        states: &RenderElementStates,
184        compare: F,
185    ) -> Option<Output>
186    where
187        F: for<'a> Fn(&'a Output, &'a RenderElementState, &'a Output, &'a RenderElementState) -> &'a Output,
188    {
189        let element_id = element_id.into();
190        let element_was_presented = states.element_was_presented(element_id.clone());
191        let element_state = states.element_render_state(element_id);
192        let primary_scanout_output = &mut self.0;
193
194        let has_valid_output = primary_scanout_output
195            .as_ref()
196            .map(|(current, _)| current.is_alive())
197            .unwrap_or(false);
198        let same_output = primary_scanout_output
199            .as_ref()
200            .map(|(current, _)| current == output)
201            .unwrap_or(false);
202
203        // If the element was not presented and we have no valid output
204        // there is nothing we can do
205        if !element_was_presented && !has_valid_output {
206            return None;
207        }
208
209        // If our current output is the one we received for the update
210        // and the element was not presented remove it, no need to check further
211        if !element_was_presented && same_output {
212            *primary_scanout_output = None;
213            return None;
214        }
215
216        // If the element was presented but we have no current valid output
217        // just directly update and exit early
218        if element_was_presented && !has_valid_output {
219            *primary_scanout_output = Some((output.downgrade(), element_state.unwrap()));
220            return Some(output.clone());
221        }
222
223        // If the element was presented on the same output there
224        // is no need to check
225        if element_was_presented && has_valid_output && same_output {
226            primary_scanout_output.as_mut().unwrap().1 = element_state.unwrap();
227            return Some(output.clone());
228        }
229
230        // If the element was presented on a different output and we have a
231        // valid current output, run the provided check
232        if element_was_presented && has_valid_output && !same_output {
233            let (current_output, current_state) = primary_scanout_output
234                .as_ref()
235                .map(|(output, state)| (output.upgrade().unwrap(), state))
236                .unwrap();
237            let updated = compare(
238                &current_output,
239                current_state,
240                output,
241                element_state.as_ref().unwrap(),
242            )
243            .clone();
244            *primary_scanout_output = Some((updated.downgrade(), element_state.unwrap()));
245            return Some(updated);
246        }
247
248        // The element was not presented on that output and the current scan-out output
249        // is a different than we currently have, so we can just leave it as is
250        primary_scanout_output
251            .as_ref()
252            .and_then(|(output, _)| output.upgrade())
253    }
254
255    /// Gets the current primary scanout output if any
256    ///
257    /// Return `None` if there is no primary scanout output or
258    /// the stored output is longer alive
259    pub fn current_output(&self) -> Option<Output> {
260        self.0.as_ref().and_then(|(o, _)| o.upgrade())
261    }
262}

264/// Default function for primary scan-out selection
265///
266/// This will prefer the next output when the visible portion of
267/// the element on screen is at least twice the size of the
268/// current visible portion. Otherwise it will prefer the output
269/// with the higher refresh rate.
270pub fn default_primary_scanout_output_compare<'a>(
271    current_output: &'a Output,
272    current_state: &RenderElementState,
273    next_output: &'a Output,
274    next_state: &RenderElementState,
275) -> &'a Output {
276    const VISIBLE_AREA_THRESHOLD: usize = 2;
277
278    let current_mode = current_output.current_mode();
279    let next_mode = next_output.current_mode();
280
281    // We don't expect to be called with an output that has no mode,
282    // but to be safe we do not unwrap here
283    let next_mode_has_higher_refresh = next_mode
284        .map(|next_mode| {
285            current_mode
286                .map(|current_mode| next_mode.refresh > current_mode.refresh)
287                .unwrap_or(false)
288        })
289        .unwrap_or(false);
290
291    let current_visible_area_threshold = current_state.visible_area * VISIBLE_AREA_THRESHOLD;
292    let next_mode_visible_area_greater = next_state.visible_area >= current_visible_area_threshold;
293
294    if next_mode_visible_area_greater || next_mode_has_higher_refresh {
295        next_output
296    } else {
297        current_output
298    }
299}

301/// Holds the states for a set of [`RenderElement`]s
302#[derive(Default, Debug, Clone)]
303pub struct RenderElementStates {
304    /// Holds the render states of the elements
305    pub states: HashMap<Id, RenderElementState>,
306}
307
308impl RenderElementStates {
309    /// Return the [`RenderElementState`] for the specified [`Id`]
310    ///
311    /// Return `None` if the element is not included in the states
312    pub fn element_render_state(&self, id: impl Into<Id>) -> Option<RenderElementState> {
313        self.states.get(&id.into()).copied()
314    }
315
316    /// Returns whether the element with the specified id was presented
317    ///
318    /// Returns `false` if the element with the id was not found or skipped
319    pub fn element_was_presented(&self, id: impl Into<Id>) -> bool {
320        self.element_render_state(id)
321            .map(|state| state.presentation_state != RenderElementPresentationState::Skipped)
322            .unwrap_or(false)
323    }
324}

326/// Defines the kind of an [`Element`]
327///
328/// This can give the backend a hint about how to handle the element
329#[derive(Debug, Copy, Clone, PartialEq, Eq, Default)]
330pub enum Kind {
331    /// The element represents a cursor
332    ///
333    /// An element representing a cursor is expected to change infrequently.
334    /// Not marking a cursor element as `Cursor` may result in lower performance and increased power usage.
335    /// In contrast, marking elements that change frequently as `Cursor` can degrade performance significantly.
336    Cursor,
337    /// The element kind is unspecified
338    #[default]
339    Unspecified,
340}

342/// A single element
343pub trait Element {
344    /// Get the unique id of this element
345    fn id(&self) -> &Id;
346    /// Get the current commit position of this element
347    fn current_commit(&self) -> CommitCounter;
348    /// Get the location relative to the output
349    fn location(&self, scale: Scale<f64>) -> Point<i32, Physical> {
350        self.geometry(scale).loc
351    }
352    /// Get the src of the underlying buffer
353    fn src(&self) -> Rectangle<f64, BufferCoords>;
354    /// Get the transform of the underlying buffer
355    fn transform(&self) -> Transform {
356        Transform::Normal
357    }
358    /// Get the geometry relative to the output
359    fn geometry(&self, scale: Scale<f64>) -> Rectangle<i32, Physical>;
360    /// Get the damage since the provided commit relative to the element
361    fn damage_since(&self, scale: Scale<f64>, commit: Option<CommitCounter>) -> DamageSet<i32, Physical> {
362        if commit != Some(self.current_commit()) {
363            DamageSet::from_slice(&[Rectangle::from_size(self.geometry(scale).size)])
364        } else {
365            DamageSet::default()
366        }
367    }
368    /// Get the opaque regions of the element relative to the element
369    fn opaque_regions(&self, _scale: Scale<f64>) -> OpaqueRegions<i32, Physical> {
370        OpaqueRegions::default()
371    }
372    /// Returns an alpha value the element should be drawn with regardless of any
373    /// already encoded alpha in it's underlying representation.
374    fn alpha(&self) -> f32 {
375        1.0
376    }
377    /// Returns the [`Kind`] for this element
378    fn kind(&self) -> Kind {
379        Kind::default()
380    }
381}

383/// A single render element
384pub trait RenderElement<R: Renderer>: Element {
385    /// Draw this element
386    fn draw(
387        &self,
388        frame: &mut R::Frame<'_, '_>,
389        src: Rectangle<f64, BufferCoords>,
390        dst: Rectangle<i32, Physical>,
391        damage: &[Rectangle<i32, Physical>],
392        opaque_regions: &[Rectangle<i32, Physical>],
393    ) -> Result<(), R::Error>;
394
395    /// Get the underlying storage of this element, may be used to optimize rendering (eg. drm planes)
396    #[inline]
397    fn underlying_storage(&self, renderer: &mut R) -> Option<UnderlyingStorage<'_>> {
398        let _ = renderer;
399        None
400    }
401}

1110/// Aggregate multiple types implementing [`RenderElement`] into a single enum type
1111///
1112/// ```
1113/// # use smithay::{
1114/// #     backend::{
1115/// #         allocator::Fourcc,
1116/// #         renderer::{
1117/// #             element::{Element, Id, RenderElement},
1118/// #             utils::CommitCounter,
1119/// #             Renderer,
1120/// #         },
1121/// #     },
1122/// #     utils::{Buffer, Point, Physical, Rectangle, Scale, Transform},
1123/// # };
1124/// #
1125/// # struct MyRenderElement1;
1126/// # struct MyRenderElement2;
1127/// #
1128/// # impl Element for MyRenderElement1 {
1129/// #     fn id(&self) -> &Id {
1130/// #         unimplemented!()
1131/// #     }
1132/// #
1133/// #     fn current_commit(&self) -> CommitCounter {
1134/// #         unimplemented!()
1135/// #     }
1136/// #
1137/// #     fn geometry(&self, _scale: Scale<f64>) -> Rectangle<i32, Physical> {
1138/// #         unimplemented!()
1139/// #     }
1140/// #
1141/// #     fn src(&self) -> Rectangle<f64, Buffer> {
1142/// #         unimplemented!()
1143/// #     }
1144/// # }
1145/// #
1146/// # impl<R: Renderer> RenderElement<R> for MyRenderElement1 {
1147/// #     fn draw(
1148/// #         &self,
1149/// #         _frame: &mut R::Frame<'_, '_>,
1150/// #         _src: Rectangle<f64, Buffer>,
1151/// #         _dst: Rectangle<i32, Physical>,
1152/// #         _damage: &[Rectangle<i32, Physical>],
1153/// #         _opaque_regions: &[Rectangle<i32, Physical>],
1154/// #     ) -> Result<(), R::Error> {
1155/// #         unimplemented!()
1156/// #     }
1157/// # }
1158/// #
1159/// # impl Element for MyRenderElement2 {
1160/// #     fn id(&self) -> &Id {
1161/// #         unimplemented!()
1162/// #     }
1163/// #
1164/// #     fn current_commit(&self) -> CommitCounter {
1165/// #         unimplemented!()
1166/// #     }
1167/// #
1168/// #     fn geometry(&self, _scale: Scale<f64>) -> Rectangle<i32, Physical> {
1169/// #         unimplemented!()
1170/// #     }
1171/// #
1172/// #     fn src(&self) -> Rectangle<f64, Buffer> {
1173/// #         unimplemented!()
1174/// #     }
1175/// # }
1176/// #
1177/// # impl<R: Renderer> RenderElement<R> for MyRenderElement2 {
1178/// #     fn draw(
1179/// #         &self,
1180/// #         _frame: &mut R::Frame<'_, '_>,
1181/// #         _src: Rectangle<f64, Buffer>,
1182/// #         _dst: Rectangle<i32, Physical>,
1183/// #         _damage: &[Rectangle<i32, Physical>],
1184/// #         _opaque_regions: &[Rectangle<i32, Physical>],
1185/// #     ) -> Result<(), R::Error> {
1186/// #         unimplemented!()
1187/// #     }
1188/// # }
1189/// use smithay::backend::renderer::element::render_elements;
1190///
1191/// render_elements! {
1192///     MyRenderElements;
1193///     First=MyRenderElement1,
1194///     Second=MyRenderElement2,
1195/// }
1196/// ```
1197///
1198/// If the [`RenderElement`] has special requirements on the [`Renderer`] you can
1199/// express them with a syntax similar to HRTBs.
1200///
1201/// For example the [`MemoryRenderBufferRenderElement`](crate::backend::renderer::element::memory::MemoryRenderBufferRenderElement) requires
1202/// the [`Renderer`] to implement the [`ImportMem`](crate::backend::renderer::ImportMem) trait.
1203///
1204/// ```
1205/// use smithay::backend::renderer::{
1206///     element::{memory::MemoryRenderBufferRenderElement, render_elements},
1207///     ImportMem,
1208/// };
1209///
1210/// render_elements! {
1211///     MyRenderElements<R> where R: ImportMem;
1212///     Memory=MemoryRenderBufferRenderElement<R>,
1213/// }
1214/// ```
1215///
1216/// In case you want to use a reference or an element with an explicit lifetime the macro
1217/// additionally takes a lifetime on the defined enum.
1218///
1219/// ```
1220/// use smithay::backend::renderer::{
1221///     element::{memory::MemoryRenderBufferRenderElement, render_elements},
1222///     ImportMem,
1223/// };
1224///
1225/// render_elements! {
1226///     MyRenderElements<'a, R> where R: ImportMem;
1227///     Memory=&'a MemoryRenderBufferRenderElement<R>,
1228/// }
1229/// ```
1230///
1231/// Additionally the macro can be used to define generic enums
1232///
1233/// ```
1234/// use smithay::backend::renderer::{
1235///     element::{memory::MemoryRenderBufferRenderElement, render_elements},
1236///     ImportMem,
1237/// };
