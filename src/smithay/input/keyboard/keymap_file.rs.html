1use std::ffi::CString;
2use std::os::unix::io::{AsFd, BorrowedFd};
3use std::sync::atomic::{AtomicUsize, Ordering};
4
5use tracing::error;
6use xkbcommon::xkb::{self, Keymap, KEYMAP_FORMAT_TEXT_V1};
7
8use crate::utils::SealedFile;
9
10/// Keymap ID, uniquely identifying the keymap without requiring a full content hash.
11static NEXT_ID: AtomicUsize = AtomicUsize::new(0);
12
13/// Wraps an XKB keymap into a sealed file or stores as just a string for sending to WlKeyboard over an fd
14#[derive(Debug)]
15pub struct KeymapFile {
16    sealed: Option<SealedFile>,
17    keymap: String,
18    id: usize,
19}
20
21impl KeymapFile {
22    /// Turn the keymap into a string using KEYMAP_FORMAT_TEXT_V1, create a sealed file for it, and store the string
23    pub fn new(keymap: &Keymap) -> Self {
24        let name = c"smithay-keymap";
25        let keymap = keymap.get_as_string(KEYMAP_FORMAT_TEXT_V1);
26        let sealed = SealedFile::with_content(name, &CString::new(keymap.as_str()).unwrap());
27
28        if let Err(err) = sealed.as_ref() {
29            error!("Error when creating sealed keymap file: {}", err);
30        }
31
32        let id = NEXT_ID.fetch_add(1, Ordering::Relaxed);
33
34        Self {
35            sealed: sealed.ok(),
36            keymap,
37            id,
38        }
39    }
40
41    #[cfg(feature = "wayland_frontend")]
42    pub(crate) fn change_keymap(&mut self, keymap: &Keymap) {
43        let keymap = keymap.get_as_string(xkb::KEYMAP_FORMAT_TEXT_V1);
44
45        let name = c"smithay-keymap-file";
46        let sealed = SealedFile::with_content(name, &CString::new(keymap.clone()).unwrap());
47
48        if let Err(err) = sealed.as_ref() {
49            error!("Error when creating sealed keymap file: {}", err);
50        }
51
52        self.id = NEXT_ID.fetch_add(1, Ordering::Relaxed);
53        self.sealed = sealed.ok();
54        self.keymap = keymap;
55    }
56
57    #[cfg(feature = "wayland_frontend")]
58    /// Run a closure with the file descriptor to ensure safety
59    pub fn with_fd<F>(&self, supports_sealed: bool, cb: F) -> Result<(), std::io::Error>
60    where
61        F: FnOnce(BorrowedFd<'_>, usize),
62    {
63        use std::{io::Write, path::PathBuf};
64
65        if let Some(file) = supports_sealed.then_some(self.sealed.as_ref()).flatten() {
66            cb(file.as_fd(), file.size());
67        } else {
68            let dir = std::env::var_os("XDG_RUNTIME_DIR")
69                .map(PathBuf::from)
70                .unwrap_or_else(std::env::temp_dir);
71
72            let mut file = tempfile::tempfile_in(dir)?;
73            file.write_all(self.keymap.as_bytes())?;
74            file.flush()?;
75
76            cb(file.as_fd(), self.keymap.len());
77        }
78        Ok(())
79    }
80
81    /// Send the keymap contained within to a WlKeyboard
82    pub fn send(
83        &self,
84        keyboard: &wayland_server::protocol::wl_keyboard::WlKeyboard,
85    ) -> Result<(), std::io::Error> {
86        use wayland_server::{protocol::wl_keyboard::KeymapFormat, Resource};
87
88        self.with_fd(keyboard.version() >= 7, |fd, size| {
89            keyboard.keymap(KeymapFormat::XkbV1, fd, size as u32);
90        })
91    }
92
93    /// Get this keymap's unique ID.
94    pub(crate) fn id(&self) -> usize {
95        self.id
96    }
97}
